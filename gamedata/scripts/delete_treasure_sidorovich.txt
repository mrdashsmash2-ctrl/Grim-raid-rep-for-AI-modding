function on_game_load()
	for i=1, 65534 do
		local se_obj = alife():object(i)
		if (se_obj) then
			local name = se_obj:name()
			local lvl_id = game_graph():vertex(se_obj.m_game_vertex_id):level_id()
			local lvl_name = lvl_id and alife():level_name(lvl_id)
			if (lvl_name == "l01_escape" and string.find(name, "level_prefix_inventory_box_0000")) then
				alife_release(se_obj)
			end
		end
	end	
end 

function on_game_start()
	RegisterScriptCallback("on_game_load", on_game_load)
end
