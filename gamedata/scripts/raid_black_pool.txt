local draw_part

function on_key_press(k)
	if k ~= DIK_keys.DIK_M then return end

	if not draw_part then
		draw_part = particles_object("amk\\raid_zomb_beee")
	end

	draw_part:play_at_pos(vector():set(db.actor:position()))
end

function on_game_start()
--	RegisterScriptCallback("on_key_press", on_key_press)
end
